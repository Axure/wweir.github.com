154 ""自动执行""""""""""""""""""""""""""""""""""""""""""""""""""""""""""""""
155 autocmd FileType * call g:compileByFileType()
156 func! g:compileByFileType()
157     if &filetype    == "c"
158         call Make()
159 ""      inoremap <expr>;    getline('.')[col('.')]==$ ? ";<CR>" : ";"
160     elseif &filetype == "cpp"
161         call Make()
162     elseif &filetype == "go"
163         call Make()
164 ""      set dictionary+=/home/wweir/.vim/go.dict
165 ""      set tags+=/home/wweir/.vim/go.tags
166         inoremap <expr>.    getline('.')[col('.')-2]=='[' && getline('.')[col('.')-1] == ']' ? "\.\.\.\<right>" : "\."
167     elseif &filetype == "python"
168         call Make()
169         inoremap <expr>.    pumvisible() && getline('.')[col('.') - 2] == "(" ? "<BS>\." : "\."
170     elseif &filetype == "html"
171         inoremap ; ;<space>
172     elseif &filetype == "Vimwiki"
173         inoremap `  <c-r>=ClosePair('`')<CR>
174         noremap <M-a>   <Esc><Esc>:VimwikiAll2HTML<CR>
175         noremap <M-b>   <Esc><Esc>:Vimwiki2HTMLBrowse<CR>
176         noremap <M-t>   <Esc><Esc>:VimwikiTable
177     elseif &filetype == "vim"
178         inoremap <  <><Left>
179         cnoremap <  <><Left>
180         inoremap >  <c-r>=ClosePair('>')<CR>
181         cnoremap >  <c-r>=ClosePair('>')<CR>
182     endif
183 endfunction

190         nnoremap <M-r>  <Esc><Esc>:w<Esc>:!chmod<Space>+x<Space>%<CR>
191     else
192         inoremap <M-r>  <Esc><Esc>:!gdb ./%<<CR>
193         nnoremap <M-r>  <Esc><Esc>:!gdb ./%<<CR>
194     endif
195     if (filereadable("Makefile") || filereadable("makefile"))
196     elseif &filetype == "c"
197         ""set makeprg=gcc\ %\ -Wall\ -std=c99\ -lm\ -g\ -o\ %<\ `pkg-config\ --cflags\ --libs\ gtk+-3.0`
198         set makeprg=gcc\ %\ -Wall\ -std=c99\ -lm\ -g\ -o\ %<

200         set makeprg=g++\ %\ -Wall\ -g\ -o\ %<

210 ""自动插入文件头
211 autocmd BufNewFile * call SetTitle()
212 fun SetTitle()
213     if &filetype == 'sh'
214         call setline(1, "\#!/bin/bash")
215     elseif &filetype == 'python'
216         call setline(1, "\#!/usr/bin/python")
217         call append(line("."), "\#-\*- coding:utf-8 -\*-")
218         call append(line(".")+1, "\#\ ".expand("%"))
219     elseif &filetype == 'Vimwiki'
220         call setline(1, "==".expand("%<")."==")
221     else
222     endif
223     normal G$
224 endf

241 ""符号配对
242 function ClosePair(char)
243     if getline('.')[col('.') - 1] == a:char
244         return "\<Right>"
245     elseif (a:char == "\'" || a:char == "\"")
246         return a:char.a:char."\<left>"
247     else
248         return a:char
249     endif
250 endf
