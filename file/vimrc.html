264 function ClosePair(char)
265     if getline('.')[col('.') - 1] == a:char
266     ¦   return "\<Right>"
267     elseif (a:char == "\'" || a:char == "\"" || a:char == "\`")
268     ¦   return a:char.a:char."\<left>"
269     else
270     ¦   return a:char
271     endif
