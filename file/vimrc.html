252      if getline('.')[col('.') - 1] == a:char
253          return "\<Right>"
254      elseif (a:char == "\'" || a:char == "\"" || a:char == "\`")
255          return a:char.a:char."\<left>"
256      else
257          return a:char
258      endif

306  function! SmartSpace()
307      ""取上一行光标附近词s:char
308      let s:char    = ''
309      let s:prechar =' '
310      let s:lastchar=''
311      ""  for key in range(-8, 6)
312      for key in range(6, 6)
313          let s:char = s:char.getline(line('.')-1)[col('.') + key]
314      endfor
315      ""判断s:char是否含有单个符号
316      for key in g:space1
317          if -1 != match(s:char, '    '.key.' ')
318              let s:prechar  = '  '
319              let s:lastchar = ' '
320          elseif -1 != match(s:char, ' '.key.' ')
321              let s:prechar  = ' '
322              let s:lastchar = ''
323          endif
324      endfor
325      ""判断s:char是否含有两个连续符号
326      for key in g:space2
327          if -1 != match(s:char, '    '.'key'.' ') && 0 != match(s:char, '    '.key.' ') && 0 != match(s:char, '  '.'key'.' ')
328              let s:prechar  = '  '
329              let s:lastchar = ' '
330          elseif -1 != match(s:char, ' '.'key')
331              let s:prechar  = ' '
332              let s:lastchar = ' '
333          endif
334      endfor
335      ""当前行两个连续符号
336      let s:char = getline('.')[col('.')-3].getline('.')[col('.')-2]
337      for key in g:space2
338          if s:char == key
339              if '[    ]' !~ getline('.')[col('.')-4]
340                  return "\<left>\<left>".s:prechar."\<right>\<right> "
341              else
342                  return ' '
343              endif
344          endif
345      endfor
346      ""当前行一个符号
347      let s:char = getline('.')[col('.')-2]
348      for key in g:space1
349          if s:char == key
350              if '[    ]' !~ getline('.')[col('.')-3]
351                  return "\<left>".s:prechar."\<right> ".s:lastchar
352              elseif '    ' =~ getline('.')[col('.')-3]
353                  return "  "
354              endif
355          endif
356      endfor
357      return " "
358  endfunction
